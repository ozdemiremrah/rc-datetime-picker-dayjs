Usage
Inline Mode
import React, {Component} from 'react';
import dayjs from 'dayjs';
import {DatetimePicker} from 'rc-datetime-picker';

class App extends Component {
  constructor() {
    super();
    this.state = {
      dayjs: dayjs()
    };
  }

  handleChange = (dayjs) => {
    this.setState({
      dayjs
    });
  }

  render() {
    return (
      <DatetimePicker
        dayjs={this.state.dayjs}
        onChange={this.handleChange}
      />
    );
  }
}

render(
  <App />,
  document.getElementById('app')
);
Popup Mode
import React, {Component} from 'react';
import dayjs from 'dayjs';
import {DatetimePickerTrigger} from 'rc-datetime-picker';

class App extends Component {
  constructor() {
    super();
    this.state = {
      dayjs: dayjs()
    };
  }

  handleChange = (dayjs) => {
    this.setState({
      dayjs
    });
  }

  render() {
    const shortcuts = {
      'Today': dayjs(),
      'Yesterday': dayjs().subtract(1, 'days'),
      'Clear': ''
    };

    return (
      <DatetimePickerTrigger
        shortcuts={shortcuts} 
        dayjs={this.state.dayjs}
        onChange={this.handleChange}>
        <input type="text" value={this.state.dayjs.format('YYYY-MM-DD HH:mm')} readOnly />
      </DatetimePickerTrigger>
    );
  }
}

render(
  <App />,
  document.getElementById('app')
);
